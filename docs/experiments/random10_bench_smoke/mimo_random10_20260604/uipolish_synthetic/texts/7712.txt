        // Delete card functionality
        document.querySelectorAll('.btn-delete').forEach(btn => {
            btn.addEventListener('click', function() {
                const card = this.closest('.card');
                if (confirm('确定要删除这个推荐吗？')) {
                    card.remove();
                    updateCount();
                }
            });
        });

        // Edit card functionality
        document.querySelectorAll('.btn-edit').forEach(btn => {
            btn.addEventListener('click', function() {
                const card = this.closest('.card');
                const title = card.querySelector('.card-title').textContent;
                alert('编辑功能：' + title);
            });
        });

        // Add card functionality
        document.getElementById('addCard').addEventListener('click', function() {
            const title = prompt('请输入影视剧名称：');
            if (title) {
                const tags = prompt('请输入标签（用逗号分隔）：');
                const desc = prompt('请输入简介：');
                
                const newCard = document.createElement('div');
                newCard.className = 'card';
                newCard.innerHTML = `
                    <div class="card-header">
                        <h3 class="card-title">${title}</h3>
                        <div class="card-actions">
                            <button class="btn-edit">编辑</button>
                            <button class="btn-delete">删除</button>
                        </div>
                    </div>
                    <div class="card-tags">
                        ${tags ? tags.split(',').map(t => `<span class="tag">${t.trim()}</span>`).join('') : ''}
                    </div>
                    <p class="card-desc">${desc || '暂无简介'}</p>
                `;
                
                document.querySelector('.cards-grid').appendChild(newCard);
                
                // Rebind events
                newCard.querySelector('.btn-delete').addEventListener('click', function() {
                    if (confirm('确定要删除这个推荐吗？')) {
                        newCard.remove();
                        updateCount();
                    }
                });
                
                newCard.querySelector('.btn-edit').addEventListener('click', function() {
                    alert('编辑功能：' + title);
                });
                
                updateCount();
            }
        });

        // Update count
        function updateCount() {
            const count = document.querySelectorAll('.card').length;
            document.querySelector('.stat-value').textContent = count;
        }

        // Keyboard shortcuts
        document.addEventListener('keydown', function(e) {
            if (e.ctrlKey && e.key === 'n') {
                e.preventDefault();
                document.getElementById('addCard').click();
            }
        });
    </script>
