  <script type="text/babel">
    const WeatherForecast = () => {
      const [city, setCity] = React.useState('');
      const [date, setDate] = React.useState('');
      const [weather, setWeather] = React.useState(null);
      const [loading, setLoading] = React.useState(false);

      const cities = [
        'New York',
        'London',
        'Tokyo',
        'Paris',
        'Sydney',
        'Beijing',
        'Dubai',
        'Singapore'
      ];

      const handleSearch = () => {
        if (!city || !date) {
          alert('Please select a city and date first.');
          return;
        }
        
        setLoading(true);
        
        // Simulate API call with setTimeout
        setTimeout(() => {
          const mockWeather = {
            city: city,
            date: date,
            temperature: Math.floor(Math.random() * 30) + 5,
            condition: ['Sunny', 'Cloudy', 'Rainy', 'Partly Cloudy'][Math.floor(Math.random() * 4)],
            humidity: Math.floor(Math.random() * 40) + 40,
            wind: Math.floor(Math.random() * 20) + 5
          };
          
          setWeather(mockWeather);
          setLoading(false);
        }, 500);
      };

      const handleUpdate = () => {
        if (!weather) {
          alert('Please search for weather first.');
          return;
        }
        
        setLoading(true);
        
        setTimeout(() => {
          setWeather({
            ...weather,
            temperature: Math.floor(Math.random() * 30) + 5,
            condition: ['Sunny', 'Cloudy', 'Rainy', 'Partly Cloudy'][Math.floor(Math.random() * 4)],
            humidity: Math.floor(Math.random() * 40) + 40,
            wind: Math.floor(Math.random() * 20) + 5
          });
          setLoading(false);
        }, 300);
      };

      const formatDate = (dateString) => {
        if (!dateString) return '';
        const options = { weekday: 'long', year: 'numeric', month: 'long', day: 'numeric' };
        return new Date(dateString).toLocaleDateString('en-US', options);
      };

      return (
        <div className="weather-container">
          <div className="header">
            <h1>Weather Forecast</h1>
          </div>

          <div className="search-controls">
            <select
              value={city}
              onChange={(e) => setCity(e.target.value)}
              aria-label="Select city"
            >
              <option value="">Select city</option>
              {cities.map((c) => (
                <option key={c} value={c}>{c}</option>
              ))}
            </select>
            
            <input
              type="date"
              value={date}
              onChange={(e) => setDate(e.target.value)}
              aria-label="Select date"
            />
            
            <button 
              className="search-btn"
              onClick={handleSearch}
              disabled={loading}
            >
              {loading ? 'Searching...' : 'Search'}
            </button>
          </div>

          {!weather && (
            <p className="instructions">
              Please select a city and date first.
            </p>
          )}

          <button 
            className="update-btn"
            onClick={handleUpdate}
            disabled={loading}
          >
            Update Weather
          </button>

          {weather && (
            <div className="weather-display">
              <div className="weather-info">
                <div className="city-name">{weather.city}</div>
                <div className="temperature">{weather.temperature}°C</div>
                <div className="condition">{weather.condition}</div>
                <div className="date-display">
                  {formatDate(weather.date)}
                </div>
                <div style={{ marginTop: '15px', fontSize: '14px', color: '#888' }}>
                  Humidity: {weather.humidity}% | Wind: {weather.wind} km/h
                </div>
              </div>
            </div>
          )}
        </div>
      );
    };

    const root = ReactDOM.createRoot(document.getElementById('root'));
    root.render(<WeatherForecast />);
  </script>
</body>
