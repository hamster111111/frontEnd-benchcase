<!DOCTYPE html>
<html lang="en">
<head>
    <meta charset="UTF-8">

    <title>Currency Converter</title>
    <script src="https://unpkg.com/react@18/umd/react.development.js"></script>
    <script src="https://unpkg.com/react-dom@18/umd/react-dom.development.js"></script>
    <script src="https://unpkg.com/@babel/standalone/babel.min.js"></script>
    <style>
        * {
            margin: 0;
            padding: 0;
            box-sizing: border-box;
        }

        body {
            font-family: -apple-system, BlinkMacSystemFont, 'Segoe UI', Roboto, sans-serif;
            background-color: #e8e8e8;
            min-height: 100vh;
            display: flex;
            justify-content: center;
            padding: 20px;
        }

        .container {
            width: 100%;
            max-width: 800px;
            padding-top: 20px;
        }

        .currency-select {
            width: 100%;
            padding: 12px 16px;
            font-size: 16px;
            border: 1px solid #ccc;
            border-radius: 4px;
            background-color: white;
            cursor: pointer;
            appearance: auto;
            margin-bottom: 30px;
        }

        .amount-input-wrapper {
            display: flex;
            justify-content: center;
            margin-bottom: 15px;
        }

        .amount-input {
            width: 220px;
            padding: 10px 14px;
            font-size: 16px;
            border: 1px solid #ccc;
            border-radius: 4px;
            background-color: white;
            text-align: left;
        }

        .amount-input::placeholder {
            color: #666;
        }

        .converted-amount-wrapper {
            display: flex;
            justify-content: center;
            margin-bottom: 20px;
        }

        .converted-amount {
            padding: 12px 24px;
            font-size: 16px;
            border: 1px solid #ccc;
            border-radius: 4px;
            background-color: white;
            min-width: 180px;
            text-align: center;
        }

        .rate-updated {
            text-align: center;

            margin-top: 10px;
        }

        .rate-value {
            font-weight: 400;
        }

<body>
    <div id="root"></div>

    <script type="text/babel">
        const { useState, useEffect } = React;

        const CurrencyConverter = () => {
            const [selectedCurrency, setSelectedCurrency] = useState('USD');
            const [amount, setAmount] = useState('');
            const [convertedAmount, setConvertedAmount] = useState('');
            const [lastUpdated, setLastUpdated] = useState('');

            // Mock exchange rates (USD as base)
            const exchangeRates = {
                'USD': 1,
                'EUR': 0.92,
                'GBP': 0.79,
                'JPY': 149.5,
                'CNY': 7.24,
                'CAD': 1.36,
                'AUD': 1.53,
                'CHF': 0.88
            };

            useEffect(() => {
                const now = new Date();
                const timeString = now.toLocaleString('en-US', {
                    month: 'short',
                    day: 'numeric',
                    year: 'numeric',
                    hour: '2-digit',
                    minute: '2-digit'
                });
                setLastUpdated(timeString);
            }, []);

            const handleAmountChange = (e) => {
                const value = e.target.value;
                setAmount(value);
                
                if (value && !isNaN(value)) {
                    const numAmount = parseFloat(value);
                    const rate = exchangeRates[selectedCurrency] || 1;
                    const converted = (numAmount * rate).toFixed(2);
                    setConvertedAmount(converted);
                } else {
                    setConvertedAmount('');
                }
            };

            const handleCurrencyChange = (e) => {
                setSelectedCurrency(e.target.value);
                if (amount && !isNaN(amount)) {
                    const numAmount = parseFloat(amount);
                    const rate = exchangeRates[e.target.value] || 1;
                    const converted = (numAmount * rate).toFixed(2);
                    setConvertedAmount(converted);
                }
            };

            return (
                <div className="container">
                    <select 
                        className="currency-select"
                        value={selectedCurrency}
                        onChange={handleCurrencyChange}
                    >
                        {Object.keys(exchangeRates).map(currency => (
                            <option key={currency} value={currency}>
                                {currency}
                            </option>
                        ))}
                    </select>

                    <div className="amount-input-wrapper">
                        <input
                            type="text"
                            className="amount-input"
                            placeholder="Enter amount"
                            value={amount}
                            onChange={handleAmountChange}
                        />
                    </div>

                    <div className="converted-amount-wrapper">
                        <div className="converted-amount">
                            Converted Amount: {convertedAmount}
                        </div>
                    </div>

                    <div className="rate-updated">
                        Rate Updated: {lastUpdated}
                    </div>
                </div>
            );
        };

        const root = ReactDOM.createRoot(document.getElementById('root'));
        root.render(<CurrencyConverter />);
    </script>
</body>
</html>
