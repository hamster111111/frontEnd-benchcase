<head>

    <title>Workflow Management Interface</title>
    <style>
        :root {
            --primary-color: #2c3e50;
            --card-bg: #ffffff;

            --status-not-started: #95a5a6;
            --status-completed: #27ae60;
        }

        body {
            font-family: -apple-system, BlinkMacSystemFont, 'Segoe UI', Roboto, 'Helvetica Neue', Arial, sans-serif;
            background-color: #f5f5f5;

        }

        .task-info {
            display: flex;

        }

        .task-status {

        .task-status.pending {

<body>
    <div class="main-container" id="app"></div>

    <script>
        // 模拟 React 组件状态
        class WorkflowBuilder {
            constructor() {
                this.tasks = [
                    { id: 1, name: 'Input Data', status: 'Pending', nextStatus: 'In Progress', buttonLabel: 'Mark as Completed' },
                    { id: 2, name: 'Validate Data', status: 'In Progress', nextStatus: 'Pending', buttonLabel: 'Mark as Completed' },
                    { id: 3, name: 'Send Notification', status: 'Not Started', nextStatus: 'In Progress', buttonLabel: 'Mark as Completed' }
                ];
                this.render();
            }

            render() {
                const container = document.getElementById('app');
                
                // Workflow Builder 部分
                let taskListHTML = '';
                this.tasks.forEach(task => {
                    const statusClass = task.status.toLowerCase().replace(' ', '-');
                    taskListHTML += `
                        <div class="task-item">
                            <div class="task-info">
                                <div class="task-name">${task.name}</div>
                                <div class="task-status ${statusClass}">${task.status}</div>
                            </div>
                            <button class="task-button" onclick="markAsCompleted(${task.id})">${task.buttonLabel}</button>
                        </div>
                    `;
                });

                // Automation Rule Editor 部分
                let ruleHTML = `
                    <div class="card rule-card">
                        <div class="card-title">Automation Rule Editor</div>
                        <div class="rule-content">
                            <div class="rule-description">If task delayed - Notify Manager</div>
                        </div>
                    </div>
                `;

                // Task Execution Dashboard 部分
                let dashboardHTML = `
                    <div class="card dashboard-card">
                        <div class="card-title">Task Execution Dashboard</div>
                        <div class="dashboard-content">
                            <div class="dashboard-info">Data Processing - Running (Active Task: Validate Data)</div>
                        </div>
                    </div>
                `;

                container.innerHTML = `
                    <div class="card">
                        <div class="card-title">Workflow Builder</div>
                        <div class="task-list">${taskListHTML}</div>
                    </div>
                    ${ruleHTML}
                    ${dashboardHTML}
                `;
            }
        }

        // 全局实例
        let workflowBuilder;

        // 标记任务为完成
        function markAsCompleted(taskId) {
            if (!workflowBuilder) return;
            
            const task = workflowBuilder.tasks.find(t => t.id === taskId);
            if (task) {
                task.status = 'Completed';
                task.nextStatus = null;
                task.buttonLabel = 'Completed';
                workflowBuilder.render();
                
                // 禁用按钮
                const buttons = document.querySelectorAll('.task-button');
                buttons.forEach(button => {
                    if (button.textContent === 'Completed') {
                        button.disabled = true;
                    }
                });
            }
        }

        // 初始化
        document.addEventListener('DOMContentLoaded', function() {
            workflowBuilder = new WorkflowBuilder();
        });
    </script>
</body>
</html>
