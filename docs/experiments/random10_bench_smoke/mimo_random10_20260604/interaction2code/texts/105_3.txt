        function showPage(page) {
            const kanbanPage = document.getElementById('kanban-page');
            const settingsPage = document.getElementById('settings-page');
            const navKanban = document.getElementById('nav-kanban');
            const navSettings = document.getElementById('nav-settings');

            if (page === 'kanban') {
                kanbanPage.style.display = 'block';
                settingsPage.style.display = 'none';
                navKanban.classList.add('active');
                navSettings.classList.remove('active');
            } else {
                kanbanPage.style.display = 'none';
                settingsPage.style.display = 'block';
                navKanban.classList.remove('active');
                navSettings.classList.add('active');
            }
        }
